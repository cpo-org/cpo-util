313   private void openActionPerformed() {
314

316     jFileChooser.setMultiSelectionEnabled(false);
317     jFileChooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
318     jFileChooser.setFileFilter(new FileFilter() {

328     if (result == JFileChooser.CANCEL_OPTION) {

336   private void openFile(File cpoMetaXml) {
337     // make sure the file is readable
338     if (cpoMetaXml == null || !cpoMetaXml.canRead()) {
339       CpoUtil.showErrorMessage("Invalid file selected.");
340       return;
341     }
342 
343     // make sure this file isn't already open
344     int tabCount = jTabbedPane.getTabCount();
345     for (int i = 0; i < tabCount; i++) {
346       CpoBrowserPanel panel = (CpoBrowserPanel)jTabbedPane.getComponentAt(i);
347       if (cpoMetaXml.equals(panel.getProxy().getCpoMetaXml())) {
348         CpoUtil.showErrorMessage("The selected file is already open: " + cpoMetaXml.getAbsolutePath());
349         return;
350       }
351     }
352 
353     addRecentFile(cpoMetaXml);
354 
355     try {
356       Proxy proxy = ProxyFactory.getInstance().getProxy(cpoMetaXml);
357       createNewBrowser(proxy);
358     } catch (CpoException ex) {
359       CpoUtil.showException(ex);
360     }
361   }

363   /**
364    * Creates a new browser tab for the supplied proxy
365    */
366   private void createNewBrowser(Proxy proxy) {
367     try {
368       CpoBrowserPanel browserPanel = new CpoBrowserPanel(proxy);
369       jTabbedPane.addTab(browserPanel.getProxy().toString(), null, browserPanel, browserPanel.getProxy().toString());
370       jTabbedPane.setSelectedComponent(browserPanel);
371 
372       JButton tabCloseButton = new JButton(closeIcon);
373       tabCloseButton.setContentAreaFilled(false);
374       tabCloseButton.setBorderPainted(false);
375       tabCloseButton.setActionCommand(Integer.toString(tabCounter++));
376       tabCloseButton.setMaximumSize(new Dimension(16, 16));
377       tabCloseButton.setMinimumSize(new Dimension(16, 16));
378       tabCloseButton.setPreferredSize(new Dimension(16, 16));
379 
380       ActionListener al = new ActionListener() {
381         public void actionPerformed(ActionEvent ae) {
382           JButton btn = (JButton)ae.getSource();
383           String s1 = btn.getActionCommand();
384           for (int i = 0; i < jTabbedPane.getTabCount(); i++) {
385             JPanel pnl = (JPanel)jTabbedPane.getTabComponentAt(i);
386             btn = (JButton)pnl.getComponent(1);
387             String s2 = btn.getActionCommand();
388             if (s1.equals(s2)) {
389               if (!checkUnsavedData("You have unsaved data, are you sure you wish to exit??", i)) {
390                 jTabbedPane.removeTabAt(i);
391               }
392               break;
393             }
394           }
395         }
396       };
397       tabCloseButton.addActionListener(al);
398 
399       // tab w/ close button
400       JPanel pnl = new JPanel(new GridBagLayout());
401       pnl.setOpaque(false);
402       pnl.add(new JLabel(browserPanel.getProxy().toString()), new GridBagConstraints(0, 0, 1, 1, 1.0, 1.0, GridBagConstraints.WEST, GridBagConstraints.BOTH, new Insets(0, 0, 0, 0), 0, 0));
403       pnl.add(tabCloseButton, new GridBagConstraints(1, 0, 1, 1, 0.0, 0.0, GridBagConstraints.NORTHEAST, GridBagConstraints.NONE, new Insets(0, 0, 0, 0), 0, 0));
404       jTabbedPane.setTabComponentAt(jTabbedPane.getTabCount() - 1, pnl);
405       jTabbedPane.setSelectedIndex(jTabbedPane.getTabCount() - 1);
406       jTabbedPane.setToolTipTextAt(jTabbedPane.getTabCount() - 1, proxy.getCpoMetaXml().getAbsolutePath());
407     } catch (CpoException ex) {
408       CpoUtil.showException(ex);
409     }
410   }

471   /**
472    * Creates a new connection
473    */
474   private void newConnection(SupportedType type) {
475     try {
476       AbstractConnectionPanel panel = type.getConnectionPanelClass().newInstance();
477       panel.newDataSourceConfig();
478       String title = "Create " + panel.getTitle();
479 
480       int result = 0;
481       boolean complete = false;
482       while (result == JOptionPane.OK_OPTION && !complete) {
483         result = JOptionPane.showConfirmDialog(this, panel, title, JOptionPane.OK_CANCEL_OPTION);
484         if (result == JOptionPane.OK_OPTION) {
485           try {
486             CtDataSourceConfig dataSourceConfig = panel.createDataSourceConfig();
487             if (!cpoUtilConfig.isSetDataConfigs()) {
488               cpoUtilConfig.addNewDataConfigs();
489             }
490             CtDataSourceConfig dsc = cpoUtilConfig.getDataConfigs().addNewDataConfig();
491             dsc.set(dataSourceConfig);
492             saveConfig();
493             complete = true;
494           } catch (CpoException ex) {
495             showErrorMessage(ex.getMessage());
496           }
497         }
498       }
499     } catch (Exception ex) {
500       showException(ex);
501     }
502   }

504   /**
505    * Edits an existing connection
506    */
507   protected void editConnection(String connectionName) {
508 
509     CtDataSourceConfig dataSourceConfig = getDataSourceConfig(connectionName);
510     if (dataSourceConfig == null) {
511       showErrorMessage("Can't find data source config for connection: " + connectionName);
512       return;
513     }
514 
515     try {
516       SupportedType type = SupportedType.getTypeForConnection(dataSourceConfig);
517       AbstractConnectionPanel panel = type.getConnectionPanelClass().newInstance();
518       panel.setDataSourceConfig(dataSourceConfig);
519       String title = "Edit " + panel.getTitle();
520 
521       int result = 0;
522       boolean complete = false;
523       while (result == JOptionPane.OK_OPTION && !complete) {
524         result = JOptionPane.showConfirmDialog(this, panel, title, JOptionPane.OK_CANCEL_OPTION);
525         if (result == JOptionPane.OK_OPTION) {
526           try {
527             CtDataSourceConfig panelDsc = panel.createDataSourceConfig();
528             for (CtDataSourceConfig dsc : cpoUtilConfig.getDataConfigs().getDataConfigArray()) {
529               // lookup based on the old name, so if it changed, we replace it
530               if (dsc.getName().equals(dataSourceConfig.getName())) {
531                 dsc.set(panelDsc);
532                 complete = true;
533                 saveConfig();
534               }
535             }
536 
537             if (!complete) {
538               showErrorMessage("Saving connection failed, could not update connection: " + connectionName);
539             }
540 
541           } catch (CpoException ex) {
542             showErrorMessage(ex.getMessage());
543           }
544         }
545       }
546     } catch (Exception ex) {
547       showException(ex);
548     }
549   }

585   /**
586    * Loads the configuration data from the CpoUtilConfig.xml file, along with the necessary properties,
587    * and bootstraping url if supplied.
588    */
589   protected void loadConfig() {
590     try {
591       // user config
592       CpoUtilConfigDocument cpoUtilConfigDocument;
593       if (!configFile.exists()) {
594         // file doesn't exist, let's make one
595         cpoUtilConfigDocument = CpoUtilConfigDocument.Factory.newInstance();
596 
597         // first, try to convert if they have the old style file
598         try {
599           cpoUtilConfig = ConvertCpoUtilLocalProperties.convert();
600         } catch (CpoException ex) {
601           showException(ex);
602         }
603 
604         // if null here, then it couldn't be converted for some reason, so just make a fresh one
605         if (cpoUtilConfig == null)  {
606           cpoUtilConfig = cpoUtilConfigDocument.addNewCpoUtilConfig();
607         }
608         saveConfig();
609       } else {
610         cpoUtilConfigDocument = CpoUtilConfigDocument.Factory.parse(configFile);
611         cpoUtilConfig = cpoUtilConfigDocument.getCpoUtilConfig();
612       }
613 
614       // read default props
615       InputStream is = getClass().getResourceAsStream("/" + CPOUTIL_PROPERTIES_FILE);
616       if (is == null) {
617         throw new IOException("Could not find properties file!\nCPU Util will exit now.");
618       }
619       props.load(is);
620 
621       // bootstrapping
622       String bootstrapUrl = props.getProperty(BOOTSTRAP_URL_PROP);
623       if (bootstrapUrl != null && !bootstrapUrl.isEmpty()) {
624         if (logger.isDebugEnabled()) {
625           logger.debug("Bootstrap Url: " + bootstrapUrl);
626         }
627         try {
628           UrlLoader loader = new UrlLoader(bootstrapUrl);
629           Thread thread = new Thread(loader);
630           thread.start();
631           thread.join(UrlLoader.TIMEOUT);
632 
633           InputStream in = loader.getInputStream();
634           if (in != null) {
635             props.load(in);
636             in.close();
637           }
638         } catch (Exception ex) {
639           logger.error("Exception caught reading bootstrap properties: " + ex);
640         }
641       }
642     } catch (MalformedURLException mue) {
643       showException(mue);
644       System.exit(1);
645     } catch (IOException ex) {
646       showException(ex);
647       System.exit(1);
648     } catch (XmlException ex) {
649       showException(ex);
650       System.exit(1);
651     }
652   }

654   /**
655    * Saves the config data
656    */
657   private void saveConfig() {
658 
659     // if the .cpoutil folder isn't there, make it
660     if (!CPOUTIL_CONFIG_DIR.exists()) {
661       CPOUTIL_CONFIG_DIR.mkdir();
662     }
663 
664     CpoUtilConfigDocument doc = CpoUtilConfigDocument.Factory.newInstance();
665     CtCpoUtilConfig config = doc.addNewCpoUtilConfig();
666     config.set(cpoUtilConfig);
667 
668     try {
669       doc.save(configFile, XmlBeansHelper.getXmlOptions());
670     } catch (Exception ex) {
671       showException(ex);
672     }
673   }

675   /**
676    * Performs a version check against the minimum version.  This is used to force the user to
677    * upgrade before using the application.
678    */
679   private void checkKillSwitch() {
680     String minimumVersion = props.getProperty(MINIMUM_VERSION);
681     if (logger.isDebugEnabled()) {
682       logger.debug("Minimum version: " + minimumVersion);
683     }
684     if (minimumVersion != null) {
685       // if it's a snapshot, strip the -SNAPSHOT
686       String version = props.getProperty(VERSION);
687       if (version.endsWith(SNAPSHOT)) {
688         version = version.substring(0, version.length() - SNAPSHOT.length());
689       }
690 
691       // if the version is something like 2.9.1, use 2.9 as the version, the .1 is a minor rev
692       int idxFirstDot = version.indexOf(".");
693       if (idxFirstDot != -1) {
694         int idxSecondDot = version.indexOf(".", idxFirstDot + 1);
695         if (idxSecondDot != -1) {
696           version = version.substring(0, idxSecondDot);
697         }
698       }
699 
700       if (logger.isDebugEnabled()) {
701         logger.debug("Version: " + version);
702       }
703 
704       double min = 0;
705       try {
706         min = Double.parseDouble(minimumVersion);
707       } catch (NumberFormatException ex) {
708         logger.error(ex.getMessage());
709       }
710

733

746       // if the set was null, it couldn't be read, so use the local copy
747       logger.debug("Couldn't connect to url, so loading local copy");

754   /**
755    * Returns the set of protected classes from the url provided.
756    * This will return null if there was a timeout issue, otherwise a set of
757    * classes will be returned.
758    *
759    * @param url The url
760    *
761    * @return the set of classes
762    */
763   private Set<String> loadProtectedClassesFromUrl(String url) {
764 
765     if (url == null) {
766       return null;
767     }
768 
769     Set<String> result = new HashSet<String>();
770     try {
771       if (logger.isDebugEnabled()) {
772         logger.debug("Url: " + url);
773       }
774       UrlLoader loader = new UrlLoader(url);
775       Thread thread = new Thread(loader);
776       thread.start();
777       thread.join(UrlLoader.TIMEOUT);
778 
779       InputStream in = loader.getInputStream();
780       if (in != null) {
781         BufferedReader br = new BufferedReader(new InputStreamReader(in));
782 
783         String inputLine;
784         while ((inputLine = br.readLine()) != null) {
785           result.add(inputLine);
786         }
787         br.close();
788         in.close();
789       }
790       return result;
791     } catch (Exception ex) {
792       if (logger.isDebugEnabled()) {
793         logger.debug("Exception caught reading from url: " + ex);
794       }
795     }
796 
797     return null;
798   }
