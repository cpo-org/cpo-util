21  package org.synchronoss.cpo.util.conversion;
22  
23  import org.slf4j.*;
24  import org.synchronoss.cpo.CpoException;
25  import org.synchronoss.cpo.core.cpoCoreConfig.*;
26  import org.synchronoss.cpo.jdbc.cpoJdbcConfig.*;
27  import org.synchronoss.cpo.util.cpoUtilConfig.*;
28  
29  import java.io.*;
30  import java.util.*;

32  /**
33   * Converts the old format of .cpoutil.properties to the new format of .cpoutil/CpoUtilConfig.xml
34   *
35   * @author michael
36   * @since 5/16/12
37   */
38  public class ConvertCpoUtilLocalProperties {
39  
40    private static Logger logger = LoggerFactory.getLogger(ConvertCpoUtilLocalProperties.class);
41  
42    private static final File oldPropFile = new File(System.getProperties().getProperty("user.home"), ".cpoutil.properties");
43  
44    private static final String JDBC_CONFIG_PROCESSOR = "org.synchronoss.cpo.jdbc.config.JdbcCpoConfigProcessor";
45    private static final String DEFAULT_META_DESCRIPTOR = "cpoutil";
46    private static final String PARAM_DELIM = ";";
47    private static final String PARAM_ASSIGNMENT = "=";
48    private static final String CUSTOM_CLASSPATH="cpoutil.classpath";
49  
50    private static final String PROP_WLSURL="cpoutil.wls.url.";
51    /*
52    private static final String PROP_WLSUSER="cpoutil.wls.user.";
53    private static final String PROP_WLSPASS="cpoutil.wls.pass.";
54    private static final String PROP_WLSCONNPOOL="cpoutil.wls.connpool.";
55    private static final String PROP_WLSINITCTXFCTRY="cpoutil.wls.initialcontextfactory.";
56    private static final String PROP_CPONAME="cpoutil.wls.cponame.";
57    private static final String PROP_THEME_URL="cpoutil.app.theme_url";
58    */
59  
60    private static final String PROP_JDBC_URL = "cpoutil.jdbc.url.";
61    private static final String PROP_JDBC_DRIVER = "cpoutil.jdbc.driver.";
62    private static final String PROP_JDBC_PARAMS = "cpoutil.jdbc.params.";
63  
64    public static CtCpoUtilConfig convert() throws CpoException {
65      logger.info("Converting CpoUtil properties file to XML...");
66  
67      logger.info("Converting: " + oldPropFile.getAbsolutePath());
68  
69      if (!oldPropFile.exists() || !oldPropFile.canRead()) {
70        // nothing to do
71        return null;
72      }
73  
74      Properties oldProps = new Properties();
75  
76      try {
77        oldProps.load(new FileInputStream(oldPropFile));
78      } catch (IOException ex) {
79        throw new CpoException(ex.getMessage(), ex);
80      }
81  
82      Set<String> servers = new TreeSet<String>();
83      Enumeration propsEnum = oldProps.propertyNames();
84      while (propsEnum.hasMoreElements()) {
85        String name = (String)propsEnum.nextElement();
86        if (name.startsWith(PROP_WLSURL)) {
87          String server = name.substring(PROP_WLSURL.length());
88          servers.add(server);
89        } else if (name.startsWith(PROP_JDBC_URL)) {
90          String server = name.substring(PROP_JDBC_URL.length());
91          servers.add(server);
92        }
93      }
94  
95      // create the document
96      CtCpoUtilConfig cpoUtilConfig = CtCpoUtilConfig.Factory.newInstance();
97  
98      // custom classpath entries
99      String customClasspath = oldProps.getProperty(CUSTOM_CLASSPATH);
100     if (customClasspath != null && !customClasspath.isEmpty()) {
101       CtCustomClasspath ctCustomClasspath = cpoUtilConfig.addNewCustomClasspath();
102       StringTokenizer st = new StringTokenizer(customClasspath, File.pathSeparator);
103       while (st.hasMoreTokens()) {
104         String entry = st.nextToken();
105         ctCustomClasspath.addClasspathEntry(entry);
106       }
107     }
108 
109     // if any servers exist
110     if (!servers.isEmpty()) {
111       CtDataConfig dataConfig = cpoUtilConfig.addNewDataConfigs();
112 
113       for (String server : servers) {
114         logger.info("Converting " + server);
115         if (oldProps.getProperty(PROP_JDBC_URL + server) == null && oldProps.getProperty(PROP_JDBC_URL + server) == null) {
116           // weblogic
117           // TODO - do we need this?
118           if (logger.isTraceEnabled()) {
119             logger.trace("Weblogic???");
120           }
121         } else {
122           // jdbc
123           String userName = null;
124           String password = null;
125           String url = oldProps.getProperty(PROP_JDBC_URL + server);
126           String driver = oldProps.getProperty(PROP_JDBC_DRIVER + server);
127           String params = oldProps.getProperty(PROP_JDBC_PARAMS + server);
128 
129           // try to parse the user/pass out of the url
130           String oraclePrefix = "jdbc:oracle:thin:";
131           String hsqlPrefix = "jdbc:hsqldb:file:";
132           if (url.startsWith(oraclePrefix)) {
133             String userPass = url.substring(oraclePrefix.length(), url.indexOf("@"));
134             userName = userPass.substring(0, userPass.indexOf("/"));
135             password = userPass.substring(userPass.indexOf("/") + 1);
136           } else if (url.startsWith(hsqlPrefix)) {
137             String userPass = url.substring(url.indexOf("user="));
138             if (userPass.contains(";")) {
139               userName = userPass.substring("user=".length(), userPass.indexOf(";"));
140               password = userPass.substring(userPass.indexOf(";")).substring("password=".length() + 1);
141             } else {
142               userName = userPass.substring("user=".length());
143               password = "";
144             }
145           }
146 
147           CtJdbcConfig jdbcConfig = CtJdbcConfig.Factory.newInstance();
148           jdbcConfig.setName(server);
149           jdbcConfig.setCpoConfigProcessor(JDBC_CONFIG_PROCESSOR);
150           jdbcConfig.setMetaDescriptorName(DEFAULT_META_DESCRIPTOR);
151 
152           CtJdbcReadWriteConfig rwc = jdbcConfig.addNewReadWriteConfig();
153           rwc.setUser(userName);
154           rwc.setPassword(password);
155           rwc.setUrl(url);
156           rwc.setDriverClassName(driver);
157 
158           if (params != null && !params.isEmpty()) {
159             StringTokenizer st = new StringTokenizer(params, PARAM_DELIM);
160             while (st.hasMoreTokens()) {
161               String token = st.nextToken();
162               StringTokenizer stNameValue = new StringTokenizer(token, PARAM_ASSIGNMENT);
163               String name = null, value = null;
164               if (stNameValue.hasMoreTokens()) {
165                 name = stNameValue.nextToken();
166               }
167               if (stNameValue.hasMoreTokens()) {
168                 value = stNameValue.nextToken();
169               }
170 
171               CtProperty prop = rwc.addNewProperty();
172               prop.setName(name);
173               prop.setValue(value);
174             }
175           }
176 
177           CtDataSourceConfig dataSourceConfig = dataConfig.addNewDataConfig();
178           dataSourceConfig.set(jdbcConfig);
179         }
180       }
181     }
182 
183     return cpoUtilConfig;
184   }
