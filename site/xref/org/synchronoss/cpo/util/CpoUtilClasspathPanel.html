21  package org.synchronoss.cpo.util;
22  
23  import org.slf4j.*;
24  
25  import javax.swing.*;
26  import java.awt.*;
27  import java.awt.event.*;
28  import java.io.File;
29  import java.util.*;
30  import java.util.List;
31  
32  public class CpoUtilClasspathPanel extends JPanel {
33  
34    private Logger logger = LoggerFactory.getLogger(this.getClass());
35  
36    // Version Id for this class
37    private static final long serialVersionUID = 1L;
38  
39    private JList jListClasspath = new JList();
40    private Vector<File> classpathEntries = new Vector<File>();
41  
42    public CpoUtilClasspathPanel(List<File> classpathEntries) {
43      this.classpathEntries.addAll(classpathEntries);
44      try {
45        jbInit();
46      } catch (Exception e) {
47        logger.error(e.getMessage(), e);
48      }
49    }
50  
51    private void jbInit() throws Exception {
52      this.setLayout(new GridBagLayout());
53  
54      JLabel jLabClasspath = new JLabel("Currently in Classpath");
55      this.add(jLabClasspath, new GridBagConstraints(0, 0, 1, 1, 0.0, 0.0, GridBagConstraints.CENTER, GridBagConstraints.NONE, new Insets(0, 0, 0, 0), 0, 0));
56  
57      jListClasspath.setListData(classpathEntries);
58      JScrollPane jScroll = new JScrollPane();
59      jScroll.getViewport().add(jListClasspath);
60      this.add(jScroll, new GridBagConstraints(0, 1, 1, 2, 1.0, 1.0, GridBagConstraints.CENTER, GridBagConstraints.NONE, new Insets(0, 0, 0, 0), 0, 0));
61  
62      JButton jButAddClasspath = new JButton("Add");
63      jButAddClasspath.addActionListener(new ActionListener() {
64        public void actionPerformed(ActionEvent e) {
65          addFile();
66        }
67      });
68      this.add(jButAddClasspath, new GridBagConstraints(1, 1, 1, 1, 0.0, 0.0, GridBagConstraints.NORTHWEST, GridBagConstraints.HORIZONTAL, new Insets(0, 0, 0, 0), 0, 0));
69  
70      JButton jButRemove = new JButton("Remove");
71      jButRemove.addActionListener(new ActionListener() {
72        public void actionPerformed(ActionEvent e) {
73          removeFile();
74        }
75      });
76      this.add(jButRemove, new GridBagConstraints(1, 2, 1, 1, 0.0, 1.0, GridBagConstraints.NORTHWEST, GridBagConstraints.HORIZONTAL, new Insets(0, 0, 0, 0), 0, 0));
77    }
78  
79    private void addFile() {
80      JFileChooser jFile = new JFileChooser();
81      jFile.setMultiSelectionEnabled(true);
82      jFile.setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
83      int result = jFile.showOpenDialog(this);
84      if (result == 1) {
85        return;
86      }
87      classpathEntries.addAll(Arrays.asList(jFile.getSelectedFiles()));
88      this.jListClasspath.setListData(classpathEntries);
89    }
90  
91    private void removeFile() {
92      Object[] selectedFiles = this.jListClasspath.getSelectedValues();
93      for (Object selectedFile : selectedFiles) {
94        classpathEntries.remove((File)selectedFile);
95      }
96      this.jListClasspath.setListData(classpathEntries);
97    }
98  
99    public List<File> getClasspathEntries() {
100     return classpathEntries;
101   }
102 }
