184   @Override
185   public void setDataSourceConfig(CtDataSourceConfig dataSourceConfig) {
186     super.setDataSourceConfig(dataSourceConfig);
187 
188     CtJdbcConfig jdbcConfig = (CtJdbcConfig)dataSourceConfig;
189 
190     jTextName.setText(jdbcConfig.getName());
191     if (jdbcConfig.isSetReadWriteConfig()) {
192       CtJdbcReadWriteConfig rwc = jdbcConfig.getReadWriteConfig();
193 
194       jTextUserName.setText(rwc.getUser());
195       jTextPassword.setText(rwc.getPassword());
196       jTextUrl.setText(rwc.getUrl());
197       if (rwc.isSetDriverClassName()) {
198         jTextDriver.setText(rwc.getDriverClassName());
199       }
200 
201       StringBuilder params = new StringBuilder();
202       for (CtProperty prop : rwc.getPropertyArray()) {
203         params.append(prop.getName());
204         params.append(PARAM_ASSIGNMENT);
205         params.append(prop.getValue());
206         params.append(PARAM_DELIM);
207       }
208       jTextAreaParams.setText(params.toString());
209     }

212   @Override
213   public CtDataSourceConfig createDataSourceConfig() throws CpoException {
214     String connectionName = jTextName.getText();
215     String userName = jTextUserName.getText();
216     String password = new String(jTextPassword.getPassword());
217     String url = jTextUrl.getText();
218     String driver = jTextDriver.getText();
219     String params = jTextAreaParams.getText();
220 
221     // validation first
222     if (connectionName == null || connectionName.isEmpty()) {
223       throw new CpoException("Connection name must be provided");
224     }
225 
226     CtJdbcConfig myConfig = getDataSourceConfig();
227 
228     // dupe?
229     CtDataSourceConfig dupe = CpoUtil.getInstance().getDataSourceConfig(connectionName);
230     if (dupe != null && !dupe.equals(myConfig)) {
231       throw new CpoException("Another connection is using the name: " + connectionName + "\n\nConnection names must be unique");
232     }
233 
234     if (userName == null || userName.isEmpty()) {
235       throw new CpoException("A user name must be provided");
236     }
237 
238     if (password.isEmpty()) {
239       throw new CpoException("A password must be provided");
240     }
241 
242     if (url == null || url.isEmpty()) {
243       throw new CpoException("A url must be provided");
244     }
245 
246     if (driver == null || driver.isEmpty()) {
247       throw new CpoException("A driver must be provided");
248     }
249 
250     // create a new data source config, so we don't muck w/ the live one
251     CtJdbcConfig jdbcConfig = newDataSourceConfig();
252     jdbcConfig.setName(connectionName);
253 
254     if (myConfig != null && myConfig.getMetaDescriptorName() != null) {
255       jdbcConfig.setMetaDescriptorName(myConfig.getMetaDescriptorName());
256     } else {
257       jdbcConfig.setMetaDescriptorName(DEFAULT_META_DESCRIPTOR);
258     }
259 
260     jdbcConfig.setCpoConfigProcessor(this.getConfigProcessor());
261 
262     CtJdbcReadWriteConfig rwc = jdbcConfig.addNewReadWriteConfig();
263     rwc.setUser(userName);
264     rwc.setPassword(password);
265     rwc.setUrl(url);
266     rwc.setDriverClassName(driver);
267 
268     if (params != null && !params.isEmpty()) {
269       StringTokenizer st = new StringTokenizer(params, PARAM_DELIM);
270       while (st.hasMoreTokens()) {
271         String token = st.nextToken();
272         StringTokenizer stNameValue = new StringTokenizer(token, PARAM_ASSIGNMENT);
273         String name = null, value = null;
274         if (stNameValue.hasMoreTokens()) {
275           name = stNameValue.nextToken();
276         }
277         if (stNameValue.hasMoreTokens()) {
278           value = stNameValue.nextToken();
279         }
280 
281         CtProperty prop = rwc.addNewProperty();
282         prop.setName(name);
283         prop.setValue(value);
284       }
285     }
286 
287     return jdbcConfig;
288   }
