34  /**
35   * Enumeration of supported meta types
36   */
37  public enum SupportedType {
38  
39    Jdbc(JdbcCpoMetaDescriptor.class, JdbcProxy.class, CtJdbcConfig.class, JdbcConnectionPanel.class),
40    Cassandra(CassandraCpoMetaDescriptor.class, CassandraProxy.class, CtCassandraConfig.class, CassandraConnectionPanel.class),
41    ;
42  
43    protected static Logger logger = LoggerFactory.getLogger(SupportedType.class);
44  
45    private Class<? extends CpoMetaDescriptor> metaDescriptorClass = null;
46    private Class<? extends Proxy> proxyClass = null;
47    private Class<? extends CtDataSourceConfig> dataSourceConfigClass = null;
48    private Class<? extends AbstractConnectionPanel> connectionPanelClass = null;
49  
50    SupportedType(Class<? extends CpoMetaDescriptor> metaDescriptorClass, Class<? extends Proxy> proxyClass, Class<? extends CtDataSourceConfig> dataSourceConfigClass, Class<? extends AbstractConnectionPanel> connectionPanelClass) {
51      this.metaDescriptorClass = metaDescriptorClass;
52      this.proxyClass = proxyClass;
53      this.dataSourceConfigClass = dataSourceConfigClass;
54      this.connectionPanelClass = connectionPanelClass;
55    }
56  
57    public Class<? extends CpoMetaDescriptor> getMetaDescriptorClass() {
58      return metaDescriptorClass;
59    }
60  
61    public Class<? extends CtDataSourceConfig> getDataSourceConfigClass() {
62      return dataSourceConfigClass;
63    }
64  
65    public Class<? extends AbstractConnectionPanel> getConnectionPanelClass() {
66      return connectionPanelClass;
67    }
68  
69    public Proxy newProxyInstance() throws Exception {
70      return proxyClass.newInstance();
71    }
72  
73    public static SupportedType getTypeForMetaDescriptor(CpoMetaDescriptor metaDescriptor) throws CpoException {
74      for (SupportedType type : values()) {
75        if (type.metaDescriptorClass.isAssignableFrom(metaDescriptor.getClass())) {
76          return type;
77        }
78      }
79      throw new CpoException("Unknown meta descriptor");
80    }
81  
82    public static SupportedType getTypeForConnection(CtDataSourceConfig dataSourceConfig) throws CpoException {
83      for (SupportedType type : values()) {
84        if (type.dataSourceConfigClass.isAssignableFrom(dataSourceConfig.getClass())) {
85          return type;
86        }
87      }
88      throw new CpoException("Unknown data source config");
89    }
90  }
