21  package org.synchronoss.cpo.util;
22  
23  import java.util.Comparator;
24  
25  /**
26   * Comparator that does a case insensitive comparison of two string values.
27   *
28   * @author Michael Bellomo
29   * @since 5/5/12
30   */
31  public class CaseInsensitiveStringComparator implements Comparator<String> {
32  
33    public CaseInsensitiveStringComparator() {
34      super();
35    }
36  
37    @Override
38    public int compare(String s1, String s2) {
39      return s1.compareToIgnoreCase(s2);
40    }
41  }
