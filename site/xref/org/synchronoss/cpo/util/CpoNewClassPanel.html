25  import javax.swing.*;
26  import java.awt.*;

28  public class CpoNewClassPanel extends JPanel {
29  
30    private static Logger logger = LoggerFactory.getLogger(CpoNewClassPanel.class);
31  
32    // Version Id for this class
33    private static final long serialVersionUID = 1L;
34    private JTextField jTextClassName = new JTextField();
35    private JCheckBox genSrcCheckBox = new JCheckBox();
36    private JTextArea jTextAreaExpression = new JTextArea();
37    private JComboBox connectionComboBox = new JComboBox();
38  
39    public CpoNewClassPanel(Proxy proxy) {
40      this(proxy, null);
41    }
42  
43    public CpoNewClassPanel(Proxy proxy, String className) {
44      try {
45        connectionComboBox.setModel(new DefaultComboBoxModel(proxy.getConnectionList()));
46        jbInit();
47      } catch (Exception e) {
48        logger.error(e.getMessage(), e);
49      }
50  
51      if (className != null) {
52        jTextClassName.setText(className);
53      } else if (proxy.getDefaultPackageName() != null) {
54        jTextClassName.setText(proxy.getDefaultPackageName());
55      }
56    }
57  
58    private void jbInit() throws Exception {
59      this.setLayout(new GridBagLayout());
60  
61      JLabel jLabClassName = new JLabel("Class Name:");
62      this.add(jLabClassName, new GridBagConstraints(0, 0, 1, 1, 0.0, 0.0, GridBagConstraints.EAST, GridBagConstraints.NONE, new Insets(2, 2, 2, 2), 0, 0));
63  
64      jTextClassName.setMinimumSize(new Dimension(300, 20));
65      jTextClassName.setPreferredSize(new Dimension(300, 20));
66      this.add(jTextClassName, new GridBagConstraints(1, 0, 1, 1, 0.0, 0.0, GridBagConstraints.WEST, GridBagConstraints.NONE, new Insets(2, 2, 2, 2), 0, 0));
67  
68      JLabel jLabGenerateSource = new JLabel("Generate Source:");
69      this.add(jLabGenerateSource, new GridBagConstraints(0, 1, 1, 1, 0.0, 0.0, GridBagConstraints.EAST, GridBagConstraints.NONE, new Insets(2, 2, 2, 2), 0, 0));
70  
71      this.add(genSrcCheckBox, new GridBagConstraints(1, 1, 1, 1, 0.0, 0.0, GridBagConstraints.WEST, GridBagConstraints.NONE, new Insets(2, 2, 2, 2), 0, 0));
72  
73      JLabel jLabConnection = new JLabel("Connection:");
74      this.add(jLabConnection, new GridBagConstraints(0, 2, 1, 1, 0.0, 0.0, GridBagConstraints.EAST, GridBagConstraints.NONE, new Insets(2, 2, 2, 2), 0, 0));
75  
76      this.add(connectionComboBox, new GridBagConstraints(1, 2, 1, 1, 0.0, 0.0, GridBagConstraints.WEST, GridBagConstraints.HORIZONTAL, new Insets(2, 2, 2, 2), 0, 0));
77  
78      JLabel jLabExpression = new JLabel("Initialization Expression:");
79      this.add(jLabExpression, new GridBagConstraints(0, 3, 1, 1, 0.0, 0.0, GridBagConstraints.NORTHEAST, GridBagConstraints.NONE, new Insets(2, 2, 2, 2), 0, 0));
80  
81      jTextAreaExpression.setLineWrap(true);
82      JScrollPane jScrollPane = new JScrollPane();
83      jScrollPane.getViewport().add(jTextAreaExpression, null);
84      jScrollPane.setPreferredSize(new Dimension(300, 200));
85      this.add(jScrollPane, new GridBagConstraints(1, 3, 1, 1, 0.0, 0.0, GridBagConstraints.WEST, GridBagConstraints.NONE, new Insets(2, 2, 2, 2), 0, 0));
86    }
